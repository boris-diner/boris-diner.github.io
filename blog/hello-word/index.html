Python Example: A Simple Sine WaveðŸ”—

import numpy as np
import matplotlib.pyplot as plt

# Time array: 0 to 1 second, 1000 samples
t = np.linspace(0, 1, 1000)
# Sine wave: 5 Hz
y = np.sin(2 * np.pi * 5 * t)

plt.plot(t, y)
plt.title("5 Hz Sine Wave")
plt.xlabel("Time (s)")
plt.ylabel("Amplitude")
plt.grid(True)
plt.show()
